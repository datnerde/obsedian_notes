- Space: O(1)
- Time: O(n)
```python
class Solution(object):
	def removeNthFromEnd(self, head, n):
		fast, slow = head, head
		# let the fast runner go n step
		for _ in range(n):
			fast = fast.next
		# check if the linked list with n length
		if not fast:
			return head.next
		while fast.next:
			fast, slow = fast.next, slow.next
		slow.next = slow.next.next
		return head
```
